# ============================================================ # HTTP / HTTPS 完全理解ガイド # ============================================================ # 【このファイルについて】 # HTTPとHTTPSの仕組みをPythonコードで実際に体験しながら学びます。 # Google ColabまたはローカルのPythonで動かしてください。 # # 【必要なライブラリのインストール】 # pip install requests # ============================================================ # ============================================================ # PART 1：HTTPとHTTPSの基礎知識 # ============================================================ # 【HTTP とは】 # HyperText Transfer Protocol の略。 # ブラウザ（クライアント）とサーバーが会話するためのルール（規約）です。 # # 例えるなら「手紙のやり取り」： # あなた（クライアント） →「このページをください」→ サーバー # サーバー →「はい、どうぞ」 → あなた # # 【HTTPS とは】 # HTTP + Security（セキュリティ）の略。 # HTTPの通信内容を「暗号化」したものです。 # # HTTPとHTTPSの違い： # HTTP → 手紙の中身が「丸見え」（第三者に盗み見られる可能性あり） # HTTPS → 手紙が「封筒に入って暗号化」（第三者には読めない） # # GitHubのサイト（nokka7465-ux.github.io）も # 「Enforce HTTPS」が有効なので、通信が暗号化されています。 # ============================================================ # PART 2：HTTPリクエストの仕組み # ============================================================ # ブラウザでURLを開くとき、裏側では以下が起きています： # # 1. DNS解決 … ドメイン名（google.com）をIPアドレスに変換 # 2. 接続 … クライアントがサーバーに接続 # 3. リクエスト … 「このページをください」と送信 # 4. レスポンス … サーバーが「200 OK」とHTMLを返す # 5. 表示 … ブラウザがHTMLを解析して画面に表示 # # 【HTTPメソッド（リクエストの種類）】 # GET … データを取得する（ページを見る） # POST … データを送信する（フォームの送信・ログイン） # PUT … データを更新する # DELETE … データを削除する # # 【HTTPステータスコード（サーバーからの返事）】 # 200 OK … 成功！ # 301 Moved … URLが変わった（リダイレクト） # 404 Not Found … ページが見つからない # 500 Server Error… サーバー側のエラー # ============================================================ # PART 3：PythonでHTTPリクエストを実際に送る # ============================================================ import requests # HTTPリクエストを簡単に送れるライブラリ import json # JSONデータを扱うライブラリ（標準ライブラリ） print("=" * 50) print("PART 3：GETリクエストを送ってみる") print("=" * 50) # requests.get() でGETリクエストを送ります。 # ブラウザでURLを開くのと同じ動作をPythonで行います。 url = "https://httpbin.org/get" # テスト用のAPIサーバー response = requests.get(url) # レスポンスの中身を確認してみましょう。 print(f"\nステータスコード: {response.status_code}") # 200なら成功 print(f"レスポンスの型: {type(response)}") # response.text でレスポンスの本文（テキスト）を取得できます。 print(f"\nレスポンス本文（最初の200文字）:") print(response.text[:200]) # ============================================================ # PART 4：レスポンスヘッダーを見る # ============================================================ # HTTPの通信には「ヘッダー」という付加情報が含まれています。 # 手紙で例えると「封筒に書いてある宛名・差出人・日付」の部分です。 print("\n" + "=" * 50) print("PART 4：レスポンスヘッダーを見る") print("=" * 50) # response.headers でヘッダー情報を取得できます。 print("\n主要なヘッダー情報:") print(f" Content-Type : {response.headers.get('Content-Type')}") # Content-Type … 返ってきたデータの種類（text/html, application/json など） print(f" Server : {response.headers.get('Server')}") # Server … どのWebサーバーソフトが動いているか print(f" Date : {response.headers.get('Date')}") # Date … レスポンスが生成された日時 # ============================================================ # PART 5：JSONデータを扱う（API連携の基本） # ============================================================ # 現代のWebサービスはJSONという形式でデータをやり取りします。 # 【JSONとは】 # JavaScript Object Notation の略。 # Pythonの辞書型（dict）とほぼ同じ形式のデータです。 # 例：{"name": "tamal", "skill": "Python"} print("\n" + "=" * 50) print("PART 5：JSONデータを取得・解析する") print("=" * 50) # 無料で使えるJSONプレースホルダーAPIからデータを取得します。 json_url = "https://jsonplaceholder.typicode.com/todos/1" json_response = requests.get(json_url) # response.json() でJSONを自動的にPythonの辞書型に変換できます。 data = json_response.json() print(f"\n取得したJSONデータ: {data}") print(f" ID : {data['id']}") print(f" タイトル: {data['title']}") print(f" 完了状態: {data['completed']}") # ============================================================ # PART 6：POSTリクエスト（データを送信する） # ============================================================ # ログインフォームや問い合わせフォームを送信するときに使います。 print("\n" + "=" * 50) print("PART 6：POSTリクエストでデータを送信する") print("=" * 50) post_url = "https://httpbin.org/post" # 送信するデータを辞書型で用意します。 send_data = { "name": "tamal", "skill": "Python", "message": "HTTPの勉強中！" } # requests.post() でPOSTリクエストを送ります。 # json= を指定するとデータをJSON形式で送信します。 post_response = requests.post(post_url, json=send_data) # 返ってきたデータからサーバーが受け取った内容を確認します。 result = post_response.json() print(f"\nサーバーが受け取ったデータ: {result['json']}") print(f"ステータスコード: {post_response.status_code}") # ============================================================ # PART 7：HTTPSの暗号化を確認する # ============================================================ # HTTPS通信では「SSL/TLS証明書」という仕組みで通信を暗号化します。 # requestsライブラリはデフォルトでHTTPS証明書を検証します。 print("\n" + "=" * 50) print("PART 7：HTTPS証明書を確認する") print("=" * 50) https_url = "https://nokka7465-ux.github.io/" https_response = requests.get(https_url) print(f"\nサイトURL : {https_url}") print(f"ステータス : {https_response.status_code}") print(f"HTTPS使用 : {https_url.startswith('https://')}") # verify=True（デフォルト）だとSSL証明書を検証します。 # verify=False にすると証明書なしでも通信できますが # セキュリティ上のリスクがあるため本番では使わないこと。 print(f"証明書検証 : 有効（デフォルト）") # ============================================================ # PART 8：エラーハンドリング（例外処理） # ============================================================ # ネットワーク通信は失敗することがあります。 # try / except でエラーを適切に処理しましょう。 print("\n" + "=" * 50) print("PART 8：エラーハンドリング") print("=" * 50) def safe_get(url): """ 安全にGETリクエストを送る関数。 エラーが起きても処理が止まらないようにします。 """ try: response = requests.get(url, timeout=5) # 5秒でタイムアウト response.raise_for_status() # 4xx/5xxエラーを例外に変換 return response except requests.exceptions.Timeout: # タイムアウト：サーバーが5秒以内に応答しなかった print(f"タイムアウト: {url}") return None except requests.exceptions.ConnectionError: # 接続エラー：URLが間違い or ネットワーク切断 print(f"接続エラー: {url}") return None except requests.exceptions.HTTPError as e: # HTTPエラー：404や500などのエラーレスポンス print(f"HTTPエラー: {e}") return None # 存在しないURLでテスト print("\n存在しないURLにアクセスしてみます...") result = safe_get("https://httpbin.org/status/404") if result is None: print("→ エラーを安全にキャッチできました！") # 正常なURLでテスト print("\n正常なURLにアクセスしてみます...") result = safe_get("https://httpbin.org/status/200") if result: print(f"→ 成功！ステータス: {result.status_code}")
